DOBOT Installation Steps

  • Download the Dobot's docker image file from OneDrive, or get a copy from me.
  • Copy '"dobotros.tar" , your Ubuntu's home directory
  • Load the docker image file, open a new terminal and run this:
sudo docker load -i dobotros.tar
  • Connect USB to your laptop
  • Allow Dobot's USB Connection to Ubuntu

Grant your Ubuntu's account to access USB

whoami


sudo usermod -a -G dialout <username>

Run Dobot's docker

  • Open a new terminal, and run:
xhost +local:docker && sudo docker run -it --net=host --ipc=host --pid=host --env="DISPLAY=$DISPLAY" --env="QT_X11_NO_MITSHM=1" --env="XAUTHORITY=$XAUTH" --volume="$XAUTH:$XAUTH"   -e LIBGL_ALWAYS_SOFTWARE=1  --device=/dev/ttyUSB0 dobotros
  • Run the Dobot's ROS package:
ros2 launch dobot_bringup dobot_magician_control_system.launch.py

**Open a new tab of terminal, and run: 

xhost +local:docker && sudo docker run -it --net=host --ipc=host --pid=host --env="DISPLAY=$DISPLAY" --env="QT_X11_NO_MITSHM=1" --env="XAUTHORITY=$XAUTH" --volume="$XAUTH:$XAUTH"   -e LIBGL_ALWAYS_SOFTWARE=1  --device=/dev/ttyUSB0 dobotros
  • Setting home position.
ros2 service call /dobot_homing_service dobot_msgs/srv/ExecuteHomingProcedure

Try these commands below:

  • Move the a goal position
ros2 action send_goal /PTP_action  dobot_msgs/action/PointToPoint "{motion_type: 1, target_pose: [150.0, 0.0, 60.0, 0.0], velocity_ratio: 0.5, acceleration_ratio: 0.3}" --feedback
  • Turn on the suction cup
ros2 service call /dobot_suction_cup_service dobot_msgs/srv/SuctionCupControl "{enable_suction: true}"
  • Turn off the suction cup
ros2 service call /dobot_suction_cup_service dobot_msgs/srv/SuctionCupControl "{enable_suction: false}"

Run Python Code

  • Open the .py file file
sudo nano dobotPythonCode.py

Insert this code:

# Import Library
import rclpy
from rclpy.node import Node
from rclpy.action import ActionClient
from dobot_msgs.action import PointToPoint
from dobot_msgs.srv import SuctionCupControl
from dobot_msgs.srv import ExecuteHomingProcedure
import time

class pick_place(Node):

    def __init__(self):  # Initiallisation
        super().__init__("pick_place")
        # Setup action client for point to point movement
        self._action_client = ActionClient(self, PointToPoint, "PTP_action")

        # Setup suction cup service for pick and place action
        self.cli = self.create_client(SuctionCupControl, "dobot_suction_cup_service")
        while not self.cli.wait_for_service(timeout_sec=1.0):
            self.get_logger().info("service not available, waiting again...")
        self.req = SuctionCupControl.Request()

        self.state = 0
        self.layer = 1
        self.position = 1
        self.x = 150.0
        self.y = 45.0
        self.z = 42.0
        self.theta = 0.0

    def send_suction_request(self, enable_suction):  # Enable/Disable Scution
        self.req.enable_suction = enable_suction
        self.suction_future = self.cli.call_async(self.req)
        self.suction_future.add_done_callback(self.suction_done_callback)

    def suction_done_callback(self, future):
        self.get_logger().info("suction done")

    def goal_response_callback(self, future):
        goal_handle = future.result()
        if not goal_handle.accepted:
            self.get_logger().info("Goal rejected :(")
            return

        self._goal_handle = goal_handle

        self.get_logger().info("Goal accepted :)")

        self._get_result_future = goal_handle.get_result_async()
        self._get_result_future.add_done_callback(self.get_result_callback)

    def get_result_callback(
        self, future
    ):  # Checksum of movement, trigger dicision() when movement is done
        result = future.result().result
        self.get_logger().info("Result: {0}".format(result))

    def feedback_callback(self, feedback):  # Feedback of current position
        self.get_logger().info(
            "Received feedback: {0}".format(feedback.feedback.current_pose)
        )

    def send_goal(self, target, mode):  # Send goal for movement
        self.get_logger().info("Waiting for action server...")
        self._action_client.wait_for_server()

        goal_msg = PointToPoint.Goal()
        goal_msg.target_pose = target
        goal_msg.motion_type = mode
        self.get_logger().info("Sending goal request...")

        self._send_goal_future = self._action_client.send_goal_async(
            goal_msg, feedback_callback=self.feedback_callback
        )

        self._send_goal_future.add_done_callback(self.goal_response_callback)

    def goto_coordinate(self, x, y, z, theta):  # Move to pickup location
        self.send_goal(target=[x, y, z, theta], mode=1)

    def pick_action(self):  # Turn on suction for pickup
        self.send_suction_request(True)

    def place_action(self):  # Turn off suction for placement
        self.send_suction_request(False)


def main(args=None):
    rclpy.init(args=args)

    action_client = pick_place()

    # move to a coordinate
    action_client.goto_coordinate(150.0, 0.0, 60.0, 0.0)
    time.sleep(1)
    # Turn on suction
    action_client.pick_action()
    time.sleep(1)
    # move to a coordinate
    action_client.goto_coordinate(150.0, 00.0, 100.0, 0.0)
    time.sleep(1)
    # Turn off suction
    action_client.place_action()
    time.sleep(1)
    
    rclpy.spin(action_client)


if __name__ == "__main__":
    main()
