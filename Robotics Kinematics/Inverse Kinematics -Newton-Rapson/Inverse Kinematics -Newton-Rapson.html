The Full Source code of Newton-Rapson (N-R): 

#! /usr/bin/env python3

import sys
import rclpy
import numpy as np
import math
import subprocess
from rclpy.node import Node


class ArmActionClient(Node):
    def __init__(self):
        super().__init__("arm_control_actionclient")

    def send_goal(self, angle1, angle2, angle3):
        print("Sending goal angles:", [angle1, angle2, angle3])
        command = [
            "ros2",
            "launch",
            "arm_robot",
            "control.launch.py",
            f"ax1:={angle1}",
            f"ax2:={angle2}",
            f"ax3:={angle3}",
        ]
        try:
            subprocess.run(command, check=True)
        except subprocess.CalledProcessError as e:
            print(f"Error launching command: {e}")

    def forward_kinematics(self, theta):
        x = (
            math.cos(theta[0])
            * (7 * math.cos(theta[1] + theta[2]) + 6 * math.cos(theta[1]))
        ) / 8
        y = (
            math.sin(theta[0])
            * (7 * math.cos(theta[1] + theta[2]) + 6 * math.cos(theta[1]))
        ) / 8
        z = (
            (51 / 40)
            - ((3 * math.sin(theta[1])) / 4)
            - (7 * math.sin(theta[1] + theta[2])) / 8
        )
        return np.array([x, y, z])

    def jacobian(self, theta):
        J = np.array(
            [
                [
                    -(
                        math.sin(theta[0])
                        * (7 * math.cos(theta[1] + theta[2]) + 6 * math.cos(theta[1]))
                    )
                    / 8,
                    -(
                        math.cos(theta[0])
                        * (7 * math.sin(theta[1] + theta[2]) + 6 * math.sin(theta[1]))
                    )
                    / 8,
                    -(7 * math.sin(theta[1] + theta[2]) * math.cos(theta[0])) / 8,
                ],
                [
                    (
                        math.cos(theta[0])
                        * (7 * math.cos(theta[1] + theta[2]) + 6 * math.cos(theta[1]))
                    )
                    / 8,
                    -(
                        math.sin(theta[0])
                        * (7 * math.sin(theta[1] + theta[2]) + 6 * math.sin(theta[1]))
                    )
                    / 8,
                    -(7 * math.sin(theta[1] + theta[2]) * math.sin(theta[0])) / 8,
                ],
                [
                    0,
                    -(7 * math.cos(theta[1] + theta[2])) / 8
                    - (3 * math.cos(theta[1])) / 4,
                    -(7 * math.cos(theta[1] + theta[2])) / 8,
                ],
            ]
        )
        return J

    def inverse_kinematics_NR(self, target, theta_init, tol=1e-6, max_iter=100):
        theta = np.array(theta_init, dtype=np.float64)
        for _ in range(max_iter):
            ee_pos = self.forward_kinematics(theta)
            error = target - ee_pos
            print("error norm:", np.linalg.norm(error))
            if np.linalg.norm(error) < tol:
                return theta
            J_pinv = np.linalg.pinv(self.jacobian(theta))
            theta += J_pinv @ error 
            # It is equal to: theta = theta+ J_pinv @ error 
            # Take note : error ==> F(x,y,z) = P(x,y,z) + P_desired(x,y,z)
        raise ValueError("Inverse Kinematics did not converge")

    def normalize_angle_rad(self, theta):
        """Normalizes an angle to be within [-pi, pi] radians."""
        theta = (theta + np.pi) % (2 * np.pi) - np.pi  # Ensure within [-pi, pi]
        return theta
    


def main():
    rclpy.init()
    action_client = ArmActionClient()
    if len(sys.argv) != 4:
        print("Usage: script.py <x> <y> <z>")
        sys.exit(1)
    try:
        x, y, z = map(float, sys.argv[1:])
    except ValueError:
        print("Invalid input. Please provide numeric values for x, y, and z.")
        sys.exit(1)
    theta_init = [0.0, -1.57, 0.0]
    target = np.array([x, y, z])
    try:
        theta_solution = action_client.inverse_kinematics_NR(target, theta_init)
        theta_solution = action_client.normalize_angle_rad(
            theta_solution
        )  # Normalize the final solution

        angle1 = float(round(math.degrees(theta_solution[0]), 2))
        angle2 = float(round(math.degrees(theta_solution[1]), 2))
        angle3 = float(round(math.degrees(theta_solution[2]), 2))

        # print("Solved Joint Angles (radians):", theta_solution)
        print(
            "Solved Joint Angles (degrees):",
            [
                angle1,
                angle2,
                angle3,
            ],
        )
        # ee_pos = action_client.forward_kinematics(theta_solution)
        angle1 = angle1 + 90.0  # offset
        angle2 = angle2 + 90.0  # offset
        angle3 = angle3

        print(
            "Adjusted Joint Angles (degrees) for offsets:",
            [
                angle1,
                angle2,
                angle3,
            ],
        )
        action_client.send_goal(angle1, angle2, angle3)
    except ValueError as e:
        print(e)
    rclpy.shutdown()


if __name__ == "__main__":
    main()
